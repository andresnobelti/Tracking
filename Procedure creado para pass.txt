--Procedure creado para encritpar pass

CREATE PROCEDURE SP_RegistrarUsuario 
	 @USERNAME VARCHAR(250)
    ,@EMAIL  VARCHAR(250) 
    ,@PASSWORD VARCHAR(MAX)
    ,@DATE date
	,@ID_ROL_FK int
AS
BEGIN
	SET NOCOUNT ON 

    INSERT INTO [USER]
           (USERNAME
           ,EMAIL
           ,PASSWORD
           ,DATE
           ,ID_ROL_FK)
    VALUES (
        @USERNAME
		,@EMAIL
        ,ENCRYPTBYPASSPHRASE('ENCRIPTAR', @PASSWORD)
        ,@DATE
        ,@ID_ROL_FK
        )
END

GO
